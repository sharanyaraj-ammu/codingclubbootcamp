      ;;

   3)

}


function roles(){
echo "$1 is a $3 of team $2"
}


function responsibility(){

case $1 in

   MI

      	for item in ${miTeam[@]}
	do
	if [ $item == "Bumrah" ]
	then
		roles $item $1 "Bowler"
	elif [ $item == "Rohit" ]
	then
		roles $item $1 "Captain"
	elif [ $item == "Ishan" ]
	then
		roles $item $1 "Batsman"
	else
		roles $item $1 "Allrounder"
	fi
	done
      	;;

   RCB

      	for item in ${rcbTeam[@]}
	do
	if [ $item == "Chahal" ]
	then
		roles $item $1 "Bowler"
	elif [ $item == "Virat" ]
	then
		roles $item $1 "Captain"
	elif [ $item == "Abd" ]
	then
		roles $item $1 "Batsman"
	else
		roles $item $1 "Allrounder"
	fi
	done

	;;

   CSK

      for item in ${cskTeam[@]}
	do
	if [ $item == "Saha" ]
	then
		roles $item $1 "Bowler"
	elif [ $item == "Dhoni" ]
	then
		roles $item $1 "Captain"
	elif [ $item == "Duplesis" ]
	then
		roles $item $1 "Batsman"
	else
		roles $item $1 "Allrounder"
	fi
	done

      ;;



   *)

     echo ""

     ;;

esac
}

function pointsEstimation(){

echo "Won $1 , Lost $lose"
echo "Points : $points"
playoffEligibility $2 $teamName
responsibility $teamName
}
